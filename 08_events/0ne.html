// ******** EVENTS (SHORT NOTES) ******** // Event = action (click, scroll, keypress, etc.) // Old way // element.onclick = function(){} // Modern way (BEST) // addEventListener(type, function, useCapture) // useCapture: // false → bubbling (child → parent) [default] // true → capturing (parent → child) // ******** EVENT OBJECT (e) ******** // e.target → actual element clicked // e.currentTarget → element where event is attached // e.type → type of event (click, etc.) // Mouse position: // e.clientX, e.clientY // e.screenX, e.screenY // Keys: // e.altKey, e.ctrlKey, e.shiftKey // ******** IMPORTANT METHODS ******** // e.preventDefault() // → stops default behavior (e.g., link opening) // e.stopPropagation() // → stops bubbling/capturing (event won't go up) // ******** EVENT DELEGATION ******** // Attach event to parent instead of multiple children // Efficient for dynamic elements document.querySelector('#images').addEventListener('click', function(e) { // Check if clicked element is IMG if (e.target.tagName === 'IMG') { // Remove clicked image let removeIt = e.target.parentNode removeIt.remove() // alternative: // removeIt.parentNode.removeChild(removeIt) } });
